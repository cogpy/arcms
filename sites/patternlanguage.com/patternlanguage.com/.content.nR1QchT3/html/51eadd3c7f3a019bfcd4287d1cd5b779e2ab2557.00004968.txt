// GROUPs give properties held in common by a number of objects
// You can create as many groups as you want.

//  Currently all of the buttons and choices (for choosing pen color, etc.)
//  are in the group normal_button. Thus they will all have the following properties
//  (except where other values are specified for particular buttons).
GROUP normal_button
  background "button grey"    // This is the color of the button.
  foreground "black"          // This is the color of the text of the button.
  font_size 11
  font helvetica
  font_style plain            // other possible choices are bold and italic
                              // one can also specify both: font_style bold italic

// This group is used for text (such as the label "pen:"
// that appears to the left of the pen width and color choices)
// that is closely associated with buttons
GROUP text_accompanying_buttons
  foreground black
  font_size 11
  font helvetica
  font_style plain
  align center               // This says that the text is to be centered on the button

// This is used for other text on button panels
GROUP instruction
  foreground black
